rgName="cookch09-rg"
registryName="cookbookcliacr"
aciName="cookcook-aci"

acrLoginServer=$(az acr show --resource-group $rgName --name $registryName --query loginServer -o tsv)

acrAdminUser=$(az acr credential show --resource-group $rgName --name $registryName --query username -o tsv)

acrAdminPass=$(az acr credential show --resource-group $rgName --name $registryName --query passwords[0].value -o tsv)

az appservice plan create --name myAppServicePlan --resource-group cookch09-rg --is-linux

az webapp create --resource-group cookch09-rg --plan myAppServicePlan --name mydockcook2551erapp --deployment-container-image-name $acrLoginServer/static-html-image:ver1.0 --docker-registry-server-user $acrAdminUser --docker-registry-server-password $acrAdminPass

appName="mydockcook2551erapp"

appHost=$(az webapp show --resource-group $rgName --name $appName --query defaultHostName -o tsv)

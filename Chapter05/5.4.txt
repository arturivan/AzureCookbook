rgName="ch05-rg"
namespaceName="sbnscookdemo"

az servicebus namespace create --resource-group $rgName --name $namespaceName --sku Standard

az servicebus queue create --resource-group $rgName --namespace-name $namespaceName --name ordersqueue

sbnsKey01=$(az servicebus namespace authorization-rule keys list --resource-group $rgName --namespace-name $namespaceName --name RootManageSharedAccessKey --query primaryKey)

endpoint="https://$namespaceName.servicebus.windows.net/ordersqueue/messages"

event='{"order": "2 large cheese pizzas"}'

curl -X POST -H "Authorization: SharedAccessSignature sr=$namespaceName.servicebus.windows.net&sig=$sbnsKey01&se=1735693200&skn=RootManageSharedAccessKey" -d "$event" $endpoint --verbose
